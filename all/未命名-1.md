class Employee:
    def __init__(self, name, employee_type):
        self.name = name
        self.employee_type = employee_type  # Full-time or Part-time
        self.health_score = 0
        self.programs = []
        self.incentives = []

    def register_program(self, program):
        self.programs.append(program)
        print(f"{self.name} has registered for {program.name}")

    def add_incentive(self, incentive):
        self.incentives.append(incentive)
        print(f"{self.name} earned incentive: {incentive}")

class Program:
    def __init__(self, name, program_type):
        self.name = name
        self.program_type = program_type  # Recreational or Health Management

class HealthManagementSystem:
    def __init__(self):
        self.employees = []
        self.programs = []

    def add_employee(self, employee):
        self.employees.append(employee)

    def add_program(self, program):
        self.programs.append(program)

    def track_participation(self, employee):
        print(f"Tracking participation for {employee.name}: {len(employee.programs)} programs registered.")

    def offer_incentive(self, employee, incentive):
        employee.add_incentive(incentive)

    def calculate_savings(self):
        total_savings = sum(30 for e in self.employees if e.employee_type == "Full-time")
        print(f"Total savings: ${total_savings} per year")

# 示例使用
# 创建系统实例
system = HealthManagementSystem()

# 添加员工
employee1 = Employee("Alice", "Full-time")
employee2 = Employee("Bob", "Part-time")
system.add_employee(employee1)
system.add_employee(employee2)

# 创建项目
program1 = Program("Soccer", "Recreational")
program2 = Program("Weight Management", "Health Management")
system.add_program(program1)
system.add_program(program2)

# 员工注册项目
employee1.register_program(program1)
employee1.register_program(program2)

# 跟踪员工参与情况
system.track_participation(employee1)

# 提供激励
system.offer_incentive(employee1, "Weight Goal Achieved")

# 计算年度节省
system.calculate_savings()
